    <script src="https://cdn.tailwindcss.com"></script>

    <style>

        .depto-card { border-left: 5px solid #3b82f6; transition: all 0.2s; }

        .depto-card:active { transform: scale(0.98); background-color: #f3f4f6; }

        .vendido { border-left-color: #ef4444; }

    </style>

</head>

<body class="bg-gray-100 p-4">

    <div class="max-w-md mx-auto">

        <h1 class="text-2xl font-bold text-center mb-2 text-blue-800">Control de Deptos</h1>

        

        <div class="flex justify-end mb-4">

            <button onclick="exportarExcel()" class="bg-blue-600 text-white px-4 py-2 rounded-lg shadow text-sm font-bold flex items-center gap-2">

                📥 Exportar Excel

            </button>

        </div>


        <div id="form-container" class="bg-white p-4 rounded-xl shadow-lg mb-6 hidden border-2 border-blue-500">

            <h2 id="edit-title" class="font-bold mb-4 text-lg">Editando Depto</h2>

            <input type="hidden" id="current-id">

            

            <label class="block text-xs font-bold text-gray-500 mb-1">COMPRADOR</label>

            <input type="text" id="comprador" class="w-full mb-3 p-2 border rounded">

            

            <label class="block text-xs font-bold text-gray-500 mb-1">VENDEDOR</label>

            <input type="text" id="vendedor" class="w-full mb-3 p-2 border rounded">

            

            <label class="block text-xs font-bold text-gray-500 mb-1">PAGOS REALIZADOS ($)</label>

            <input type="number" id="pagos" class="w-full mb-3 p-2 border rounded">

            

            <label class="block text-xs font-bold text-gray-500 mb-1">NOTAS</label>

            <textarea id="notas" class="w-full mb-4 p-2 border rounded h-20"></textarea>

            

            <div class="flex gap-2">

                <button onclick="guardar()" class="bg-green-600 text-white flex-1 p-3 rounded-lg font-bold">Guardar</button>

                <button onclick="cerrarForm()" class="bg-gray-400 text-white flex-1 p-3 rounded-lg font-bold">Cancelar</button>

            </div>

        </div>


        <div id="lista" class="grid gap-3"></div>

    </div>


    <script>

        // Inicializar datos

        let departamentos = JSON.parse(localStorage.getItem('deptos')) || 

            Array.from({length: 16}, (_, i) => ({

                id: i + 1, comprador: '', vendedor: '', pagos: 0, notas: ''

            }));


        function render() {

            const lista = document.getElementById('lista');

            lista.innerHTML = '';

            

            departamentos.forEach(d => {

                const isVendido = d.comprador && d.comprador.trim() !== "";

                lista.innerHTML += `

                    <div onclick="editar(${d.id})" class="depto-card ${isVendido ? 'vendido' : ''} bg-white p-4 rounded shadow flex justify-between items-center cursor-pointer">

                        <div>

                            <span class="font-bold text-blue-600">Depto ${d.id}</span>

                            <p class="text-sm text-gray-500">${isVendido ? d.comprador : 'Disponible'}</p>

                        </div>

                        <div class="text-right text-sm">

                            <p class="font-bold text-green-700">$${d.pagos}</p>

                            <span class="text-[10px] uppercase text-gray-400 font-bold italic">Tocar para editar</span>

                        </div>

                    </div>

                `;

            });

            // Guardar en memoria del teléfono

            localStorage.setItem('deptos', JSON.stringify(departamentos));

        }


        function editar(id) {

            const d = departamentos.find(x => x.id === id);

            document.getElementById('current-id').value = id;

            document.getElementById('edit-title').innerText = `Editando Depto ${id}`;

            document.getElementById('comprador').value = d.comprador;

            document.getElementById('vendedor').value = d.vendedor;

            document.getElementById('pagos').value = d.pagos;

            document.getElementById('notas').value = d.notas;

            

            document.getElementById('form-container').classList.remove('hidden');

            window.scrollTo({top: 0, behavior: 'smooth'});

        }


        function guardar() {

            const id = parseInt(document.getElementById('current-id').value);

            const index = departamentos.findIndex(x => x.id === id);

            

            departamentos[index] = {

                id: id,

                comprador: document.getElementById('comprador').value,

                vendedor: document.getElementById('vendedor').value,

                pagos: document.getElementById('pagos').value || 0,

                notas: document.getElementById('notas').value

            };

            

            cerrarForm();

            render();

        }


        function cerrarForm() {

            document.getElementById('form-container').classList.add('hidden');

        }


        function exportarExcel() {

            let contenido = "Departamento,Comprador,Vendedor,Pagos,Notas\n";

            departamentos.forEach(d => {

                const notaLimpia = d.notas.replace(/,/g, ".").replace(/\n/g, " ");

                contenido += `${d.id},${d.comprador},${d.vendedor},${d.pagos},${notaLimpia}\n`;

            });


            const blob = new Blob(["\ufeff" + contenido], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            const url = URL.createObjectURL(blob);

            link.href = url;

            link.setAttribute("download", "ventas_inmobiliaria.csv");

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }


        // Carga inicial

        render();

    </script>

</body>

</html>
